using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ATM
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Enter The Amount for Withdrawal");
            int withdrawalAmount = Convert.ToInt16(Console.ReadLine());
            

            if (withdrawalAmount == 0)
            {
                Console.WriteLine("Please Enter the Amount");
            }
            else
            {
                Console.WriteLine();
                if (withdrawalAmount % 5 == 0)
                {
                    int fiveHundred = withdrawalAmount / 500;
                    double fiveHundred_Remaining = withdrawalAmount % 500;
                    int fiveHundredRemaining_Int = Convert.ToInt16(fiveHundred_Remaining);
                    if (fiveHundred > 0)
                    {
                        Console.WriteLine("500 * " + fiveHundred);
                    }
                    withdrawalAmount = fiveHundredRemaining_Int;


                    int Hundred = withdrawalAmount / 100;
                    if (Hundred > 0)
                    {
                        Console.WriteLine("100 * " + Hundred);
                        double hundred_Remaining = withdrawalAmount % 100;
                        int hundredRemaining_Int = Convert.ToInt16(hundred_Remaining);
                        withdrawalAmount = hundredRemaining_Int;
                    }


                    int fifty = Convert.ToInt32(withdrawalAmount / 50);
                    if (fifty > 0)
                    {
                        Console.WriteLine("50 * " + fifty);
                        double fifty_Remaining = withdrawalAmount % 50;
                        int fiftyRemaining_Int = Convert.ToInt16(fifty_Remaining);
                        withdrawalAmount = fiftyRemaining_Int;
                    }



                    int Ten = Convert.ToInt32(withdrawalAmount / 10);

                    if (Ten > 0)
                    {
                        Console.WriteLine("10 * " + Ten);
                        double Ten_Remaining = withdrawalAmount % 10;
                        int Ten_Remaining_Int = Convert.ToInt16(Ten_Remaining);
                        withdrawalAmount = Ten_Remaining_Int;
                    }



                    int five = Convert.ToInt32(withdrawalAmount / 5);

                    if (five > 0)
                    {
                        Console.WriteLine("5 * " + five);
                       
                    }
                }
                else
                {
                    Console.WriteLine("Invalid Amount");
                }
               
                
                Console.ReadLine();
            }

        }
    }
}
